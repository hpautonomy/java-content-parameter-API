6   package com.hp.autonomy.aci.content.sort;
7   
8   import java.util.Collection;
9   import java.util.Collections;
10  import java.util.Iterator;
11  import java.util.LinkedHashSet;
12  import java.util.Set;
13  
14  /**
15   * A builder for sort expressions. The order of sort specifiers within the builder is significant but the specifiers are
16   * deduplicated.
17   */
18  public class SortBuilder extends AbstractSort {
19      private final Set<SortBy> sorts = new LinkedHashSet<SortBy>();
20  
21      /**
22       * Creates a new {@code SortBuilder} with the sort specifiers provided. Specifiers are added in the order they are
23       * encountered.
24       *
25       * @param sorts Initial sort specifiers
26       */
27      public SortBuilder(final Iterable<? extends SortBy>... sorts) {
28          doThen(sorts);
29      }
30  
31      /**
32       * Appends the sort specifiers provided to the current {@code SortBuilder} in the order they are encountered.
33       *
34       * @param sorts Sort specifiers
35       * @return {@code this}
36       */
37      @Override
38      public SortBuilder then(final Iterable<? extends SortBy>... sorts) {
39          return doThen(sorts);
40      }
41      
42      private SortBuilder doThen(final Iterable<? extends SortBy>... sorts) {
43          if(sorts == null) {
44              throw new IllegalArgumentException("The sort iterable provided was null");
45          }
46  
47          // In case of nulls, build a new Set and then add them all
48          final Collection<SortBy> newSorts = new LinkedHashSet<SortBy>();
49          
50          for(final Iterable<? extends SortBy> sortIterable : sorts) {
51              if(sortIterable == null) {
52                  throw new IllegalArgumentException("One of the sort iterables provided was null");
53              }
54  
55              for(final SortBy sort : sortIterable) {
56                  if(sort == null) {
57                      throw new IllegalArgumentException("One of the sort types provided was null");
58                  }
59  
60                  newSorts.add(sort);
61              }
62          }
63  
64          this.sorts.addAll(newSorts);
65          
66          return this;      
67      }
68  
69      /**
70       * Creates an iterator for the current state of the {@code SortBuilder}. Removing elements is not permitted and any
71       * changes to the builder while iterating could cause a
72       * {@link java.util.ConcurrentModificationException ConcurrentModificationException} to be
73       * thrown.
74       *
75       * @return An iterator for the sort specifiers
76       */
77      @Override
78      public Iterator<SortBy> iterator() {
79          return Collections.unmodifiableCollection(sorts).iterator();
80      }
81  
82      /**
83       * The number of specifiers currently in the builder.
84       *
85       * @return The size of the builder
86       */
87      @Override
88      public int size() {
89          return sorts.size();
90      }
91  
92      /**
93       * Whether or not {@code size() == 0}.
94       *
95       * @return {@inheritDoc}
96       */
97      @Override
98      public boolean isEmpty() {
99          return sorts.isEmpty();
100     }
101 
102     /**
103      * Converts an iterable of sort values into a {@code SortBuilder}. This method can be more efficient than using
104      * the equivalent constructor but the returned object might be backed by the object provided.
105      *
106      * @param sorts The print values to convert
107      * @return An equivalent instance of {@code PrintBuilder}
108      */
109     public static SortBuilder from(final Iterable<? extends SortBy> sorts) {
110         return (sorts instanceof SortBuilder) ? (SortBuilder)sorts : new SortBuilder(sorts);
111     }
112 }
