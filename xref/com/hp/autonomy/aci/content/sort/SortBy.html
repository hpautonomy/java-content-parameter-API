15  /**
16   * Class to represent the various sort specifiers. e.g.
17   *
18   * <pre> 
19   * String sort = SortBy.RELEVANCE
20   *               .then(SortBy.alphabetical("VIRTUALPATHNAME"))
21   *               .then(SortBy.AUTN_RANK)
22   *               .toString();
23   * </pre>
24   *
25   * Instances are accessed via static fields or methods. All instances of {@code SortBy} should be immutable, though this
26   * class is not {@code final} to allow extensions for sort specifiers that do not currently exist.
27   */
28  public class SortBy extends AbstractSort {
29      private final String sort;
30      
31      public static final SortBy AUTN_RANK = new SortBy("autnrank");
32      public static final SortBy CLUSTER = new SortBy("cluster");
33      public static final SortBy DATABASE = new SortBy("database");
34      public static final SortBy DATE = new SortBy("date");
35      public static final SortBy DOC_ID_DECREASING = new SortBy("dociddecreasing");
36      public static final SortBy DOC_ID_INCREASING = new SortBy("docidincreasing");
37      public static final SortBy OFF = new SortBy("off");
38      public static final SortBy RANDOM = new SortBy("random");
39      public static final SortBy RELEVANCE = new SortBy("relevance");
40      public static final SortBy REVERSE_DATE = new SortBy("reversedate");
41      public static final SortBy REVERSE_RELEVANCE = new SortBy("reverserelevance");
42  
43      /**
44       * Constructor for creating a {@code SortBy} that only requires a specifier name.
45       * <p>
46       * This constructor is {@code protected} to discourage its use, not because a subclass is strictly required. All of
47       * the current sort specifiers of this form exist as immutable, static fields of this class.
48       *
49       * @param sort The specifier name
50       */
51      protected SortBy(final String sort) {
52          Validate.isTrue(StringUtils.isNotBlank(sort), "Sort must not be blank");
53          this.sort = sort;
54      }
55  
56      /**
57       * Constructor for creating a {@code SortBy} that requires both a specifier name and a field name.
58       * <p>
59       * This constructor is {@code protected} to discourage its use, not because a subclass is strictly required. All of
60       * the current sort specifiers of this form can be created via static methods of this class.
61       *
62       * @param sort The specifier name
63       * @param fieldname The field name
64       */
65      protected SortBy(final String sort, final String fieldname) {
66          Validate.isTrue(StringUtils.isNotBlank(sort), "Sort must not be blank");
67          Validate.isTrue(StringUtils.isNotBlank(fieldname), "Fieldname must not be blank");
68  
69          this.sort = AciURLCodec.getInstance().encode(fieldname) + ':' + sort;
70      }
71  
72      /**

74       *
75       * @param fieldname The field name
76       * @return A new specifier
77       */
78      public static SortBy alphabetical(final String fieldname) {
79          return new SortBy("alphabetical", fieldname);
80      }
