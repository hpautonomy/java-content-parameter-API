7   /**
8    * {@code QueryIdentifiers} are document identifiers that can be used to filter a <tt>query</tt> action using a 'match'
9    * parameter, such as <tt>matchreference</tt> or <tt>statematchid</tt>. e.g.:
10   *
11   * <pre>
12   *    QueryIdentifiers identifiers = ...;
13   *
14   *    AciParameters parameters = new AciParameters("query");
15   *    parameters.add("text", "*");
16   *    parameters.add(identifiers.getMatchParameterName(), identifiers);
17   * </pre>
18   */
19  public interface QueryIdentifiers extends IndexingIdentifiers {
20      String DATABASE_MATCH = "databasematch";
21      String MATCH_ID = "matchid";
22      String MATCH_REFERENCE = "matchreference";
23      String STATE_MATCH_ID = "statematchid";
24  
25      /**
26       * The name of the 'match' parameter to use for this set of identifiers, such as <tt>matchreference</tt>.
27       *
28       * @return The <tt>query</tt> 'match' parameter name
29       */
30      String getMatchParameterName();
31  
32      /**
33       * The string representation of this identifier, suitable for use as the value of a 'match' parameter on a
34       * <tt>query</tt> action.
35       *
36       * @return The <tt>query</tt> 'match' parameter value
37       */
38      @Override
39      String toString();
40  }
