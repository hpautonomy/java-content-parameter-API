5   package com.hp.autonomy.aci.content.identifier.stateid;

7   import com.hp.autonomy.aci.content.identifier.DocumentIdentifiers;
8   
9   /**
10   * A representation of a set of state ids. A state id identifies a set of documents within a stored state in IDOL. A
11   * stored state is primarily identified by an opaque string known as a state token but a state id can also include a
12   * zero-based range of indices that restrict the state id to a subset of the documents in the stored state. e.g.
13   * <tt>TOKEN[3-6]</tt>. Such ranges are represented by the {@link StateRange} class.
14   */
15  public interface StateIds extends DocumentIdentifiers, Iterable<StateId> {
16  
17      /**
18       * Combines the state ids in {@code this} with the specified state id and zero-based document index.
19       * <p>
20       * It is implementation specific whether or not {@code this} is modified or whether a new object is returned
21       * instead. If {@code this} is modified it should also be returned.
22       *
23       * @param stateToken The state token to append
24       * @param document The document index for the state token
25       * @return The combined state ids object
26       */
27      StateIds append(String stateToken, int document);
28  
29      /**
30       * Combines the state ids in {@code this} with the specified state id and zero-based document index range.
31       * <p>
32       * It is implementation specific whether or not {@code this} is modified or whether a new object is returned
33       * instead. If {@code this} is modified it should also be returned.
34       *
35       * @param stateToken The state token to append
36       * @param start The start index of the range
37       * @param end The end index of the range
38       * @return The combined state ids object
39       */
40      StateIds append(String stateToken, int start, int end);
41  
42      /**
43       * Combines the state ids in {@code this} with the specified state id and range.
44       * <p>
45       * It is implementation specific whether or not {@code this} is modified or whether a new object is returned
46       * instead. If {@code this} is modified it should also be returned.
47       *
48       * @param stateToken The state token to append
49       * @param range The index range
50       * @return The combined state ids object
51       */
52      StateIds append(String stateToken, StateRange range);
53  
54      /**
55       * Combines the state ids in {@code this} with the specified state tokens.
56       * <p>
57       * It is implementation specific whether or not {@code this} is modified or whether a new object is returned
58       * instead. If {@code this} is modified it should also be returned.
59       *
60       * @param stateTokens The state tokens to append
61       * @return The combined state ids object
62       */
63      StateIds append(String... stateTokens);
64  
65      /**
66       * Combines the state ids in {@code this} with the specified state ids. The {@link Iterable} can contain instances
67       * of {@link String} or {@link StateId}. Note that all instances of {@link StateIds} are suitable iterables.
68       * <p>
69       * It is implementation specific whether or not {@code this} is modified or whether a new object is returned
70       * instead. If {@code this} is modified it should also be returned.
71       *
72       * @param stateIds The state ids to append
73       * @return The combined state ids object
74       */
75      StateIds append(Iterable<?>... stateIds);
76  
77      /**
78       * The number of state ids.
79       *
80       * @return The size
81       */
82      int size();
83  
84      /**
85       * {@code StateIds} objects are considered equal if their {@code String} representations are equal.
86       *
87       * @param obj An object to test for equality
88       * @return {@code true} if and only if {@code obj} is a {@code StateIds} object with the same {@code toString()}
89       *         value as this object.
90       */
91      @Override
92      boolean equals(Object obj);
93  
94      /**
95       * The hashcode should be that of the {@code String} representation.
96       *
97       * @return The hashcode of the {@code String} representation.
98       */
99      @Override
100     int hashCode();
101 }
