5   package com.hp.autonomy.aci.content.identifier.stateid;
6   
7   /**
8    * A representation of the state range for a state id. The state range is the optional section in square brackets after
9    * the state token.
10   *
11   * <p>This interface provides relatively few methods for manipulating a state range. This is intentional. Though the
12   * majority of ranges are continuous it is possible to have discontinuous ranges and these must also be supported.
13   */
14  public interface StateRange {
15      /**
16       * The string representation of the range, as used to query IDOL. Note that this should not include the square
17       * brackets. An empty range should have an empty string representation.
18       *
19       * @return The range as a string
20       */
21      @Override
22      String toString();
23  
24      /**
25       * {@code StateRange} objects are considered equal if their {@code String} representations are equal.
26       *
27       * @param obj An object to test for equality
28       * @return {@code true} if and only if {@code obj} is a {@code StateRange} object with the same {@code toString()}
29       *         value as this object.
30       */
31      @Override
32      boolean equals(Object obj);
33  
34      /**
35       * The hashcode should be that of the {@code String} representation.
36       *
37       * @return The hashcode of the {@code String} representation.
38       */
39      @Override
40      int hashCode();
41  
42      /**
43       * Whether or not the range is empty. An empty range will have a size of 0 and an empty string representation.
44       *
45       * <p>In general, implementations should be wary of allowing empty ranges to be created as they can cause confusion.
46       * Despite having a size of 0, an empty range will actually match all documents in a stored state, rather than none
47       * of them.
48       *
49       * @return {@code true} if the range is empty
50       */
51      boolean isEmpty();
52  
53      /**
54       * The size of the range. For example, the range <tt>[5-7,12]</tt> has a size of {@code 4}. Note that a size of 0
55       * signifies a range that will match all the documents in a stored state rather than one that will match no
56       * documents.
57       *
58       * @return The size of the range
59       */
60      int size();
61  }
