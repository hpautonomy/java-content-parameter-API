7   /**
8    * {@code IndexingIdentifiers} are document identifiers that can be used in the data of a <tt>DREREPLACE</tt>. e.g.:
9    *
10   * <pre>
11   *    IndexingIdentifiers identifiers = ...;
12   *
13   *    // Build up the first line of the replace body
14   *    StringBuilder dreReplace = new StringBuilder()
15   *            .append('#')
16   *            .append(identifiers.getIndexingIdentifierName())
17   *            .append(' ')
18   *            .append(identifiers.toIndexingString())
19   *            .append('\n');
20   * </pre>
21   */
22  public interface IndexingIdentifiers {
23  
24      String DREALL = "DREALL";
25      String DREDOCID = "DREDOCID";
26      String DREDBNAME = "DREDBNAME";
27      String DREDOCREF = "DREDOCREF";
28      String DRESTATEID = "DRESTATEID";
29  
30      /**
31       * {@code IndexingIdentifier} to represent <tt>DREALL</tt>, which can be reused to replace fields on all the
32       * documents in an engine. This might be used when purging a particular field-value pair from an entire engine
33       * without the need to explicitly find which documents have that value.
34       *
35       * <p>This object is immutable.
36       */
37      IndexingIdentifiers ALL = new IndexingIdentifiers() {
38          @Override
39          public String toIndexingString() {
40              return "";
41          }
42  
43          @Override
44          public String getIndexingIdentifierName() {
45              return DREALL;
46          }
47  
48          @Override
49          public boolean isEmpty() {
50              return false;
51          }
52      };
53  
54      /**
55       * Whether or not the identifier is empty.
56       *
57       * <p>An empty identifier is one that cannot correspond to any documents, e.g. an empty reference list. Under some
58       * circumstances it can be unsafe to use an empty identifier as it could match all documents, rather than no
59       * documents. Though the classes in this API attempt to cope with this contingency as smoothly as possible (using
60       * dummy values and throwing exceptions as appropriate), it is usually preferable to do an empty check prior to
61       * using an identifier and handling that case separately.
62       *
63       * @return {@code true} if the identifier is empty
64       */
65      boolean isEmpty();
66  
67      /**
68       * The string representation of this identifier, suitable for use as part of a <tt>DREREPLACE</tt>.
69       *
70       * <p> Though the {@code IndexingIdentifiers} interface does not prescribe an implementation of the
71       * {@link Object#toString()} method, in practice it usually gives an ACI-ready version of the identifier. In most
72       * cases this will be the same as the indexing version but care should be taken to use the correct method as
73       * differences between the strings can cause subtly incorrect behaviour.
74       *
75       * @return A <tt>DREREPLACE</tt> identifier
76       */
77      String toIndexingString();
78  
79      /**
80       * The identifier name, as used in <tt>DREREPLACE</tt>. It will not include the leading #. e.g. <tt>DREDOCREF</tt>.
81       *
82       * @return A <tt>DREREPLACE</tt> identifier name
83       */
84      String getIndexingIdentifierName();
85  }
