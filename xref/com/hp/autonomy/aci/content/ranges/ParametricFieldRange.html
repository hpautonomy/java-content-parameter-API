6   package com.hp.autonomy.aci.content.ranges;
7   
8   import org.apache.commons.collections4.CollectionUtils;
9   
10  import java.util.ArrayList;
11  import java.util.List;
12  import java.util.regex.Matcher;
13  import java.util.regex.Pattern;
14  
15  public abstract class ParametricFieldRange<T> {
16      private static final Pattern FIELD_SEPARATOR = Pattern.compile(":", Pattern.LITERAL);
17      private final String field;
18      private final List<T> ranges;
19      private final boolean noMin;
20      private final boolean noMax;
21  
22      ParametricFieldRange(final String field, final List<T> ranges) {
23          this(field, ranges, false);
24      }
25  
26      ParametricFieldRange(final String field, final List<T> ranges, final boolean noMax) {
27          this(field, ranges, false, noMax);
28      }
29  
30      ParametricFieldRange(final String field, final List<T> ranges, final boolean noMin, final boolean noMax) {
31          this.field = field;
32          this.ranges = new ArrayList<>(ranges);
33          this.noMin = noMin;
34          this.noMax = noMax;
35      }
36  
37      protected abstract String getFieldTextName();
38  
39      protected abstract String formatElement(final T range);
40  
41      @Override
42      public String toString() {
43          if(CollectionUtils.isNotEmpty(ranges)) {
44              final StringBuilder stringBuilder = (new StringBuilder(getFieldTextName())).append("{");
45              if(noMin) {
46                  stringBuilder.append(".,");
47              }
48              ranges.forEach(range -> stringBuilder.append(formatElement(range)).append(','));
49              stringBuilder.deleteCharAt(stringBuilder.length() - 1);
50              if(noMax) {
51                  stringBuilder.append(",.");
52              }
53              stringBuilder.append("}:").append(FIELD_SEPARATOR.matcher(field).replaceAll(Matcher.quoteReplacement("%3A")));
54              return stringBuilder.toString();
55          } else {
56              return "";
57          }
58      }
59  }
