6   package com.hp.autonomy.aci.content.ranges;

8   import java.time.ZonedDateTime;
9   import java.time.format.DateTimeFormatter;
10  import java.time.temporal.ChronoUnit;
11  import java.util.List;
12  
13  @SuppressWarnings("WeakerAccess")
14  public class DateRange extends ParametricFieldRange<ZonedDateTime> {
15      private static final long serialVersionUID = -5755223676715240483L;
16  
17      public DateRange(final String field, final List<ZonedDateTime> ranges) {
18          super(field, ranges);
19      }
20  
21      public DateRange(final String field, final List<ZonedDateTime> ranges, final boolean noMax) {
22          super(field, ranges, noMax);
23      }
24  
25      public DateRange(final String field, final List<ZonedDateTime> ranges, final boolean noMin, final boolean noMax) {
26          super(field, ranges, noMin, noMax);
27      }
28  
29      protected String getFieldTextName() {
30          return "FIXEDDATE";
31      }
32  
33      protected String formatElement(final ZonedDateTime range) {
34          // IDOL ISO date time does not have a year 0 whereas Java ISO date time does
35          return DateTimeFormatter.ISO_INSTANT
36              .format(range.truncatedTo(ChronoUnit.SECONDS));
37      }
38  }
