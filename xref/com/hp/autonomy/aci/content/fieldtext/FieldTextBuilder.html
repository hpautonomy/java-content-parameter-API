12  /**
13   * <p>{@code FieldTextBuilder} allows for the building of complex fieldtext expressions from other fieldtext expressions
14   * or specifiers. e.g.:
15   * <p>
16   * <pre>
17   *    FieldTextBuilder builder = new FieldTextBuilder(new MATCH("MYFIELD", "VALUE"));
18   *    builder.AND(new EQUAL("NUMERIC_FIELD", 6));</pre>
19   * <p>
20   * evaluates to:
21   * <p>
22   * <pre>    MATCH{VALUE}:MYFIELD+AND+EQUAL{6}:NUMERIC_FIELD</pre>
23   *
24   * Note that calls to {@code AND}, {@code OR}, {@code XOR} and {@code NOT} return the same builder on which they were
25   * called so the return value is usually discarded.
26   *
27   * <p>{@code FieldTextBuilder} manages boolean operators to ensure that there are no leading or trailing operators.
28   * Further, it will omit parentheses where possible.
29   *
30   * <p>This class is {@code final} to ensure that the assumptions made in the optimizations for combining
31   * {@code FieldTextBuilder}s cannot be violated.
32   *
33   * <p>This class makes no attempt at internal synchronization and is not safe to be used by multiple threads without
34   * external synchronization.
35   *
36   * <p>The lack of an empty constructor is not an oversight, it caused some confusion in earlier versions of the class
37   * and has been removed. The problem occurred in the treatment of the OR operator, which is logically expected to
38   * increase the size of the results set. However, an empty fieldtext matches all documents so appending more fieldtext
39   * via OR should have no effect. This was not the behaviour in earlier versions of this class and this counter-intuitive
40   * behaviour led to problems. The problem was fixed using the {@link FieldTexts#MATCHNOTHING} specifier but introduced
41   * another problem in that it was now much more difficult to build up a sequence of expressions separated by OR. Static
42   * factory methods have been added to make this easier, see the examples below for current usage patterns.
43   *
44   * <p>Example 1 - Some of the fieldtext is fixed and can be added at construction time:
45   *
46   * <pre>
47   *    FieldTextBuilder builder = new FieldTextBuilder(new MATCH("FIELD", "VALUE"));</pre>
48   * <p>
49   * <p>or:
50   * <p>
51   * <pre>
52   *    FieldTextBuilder builder = FieldTextBuilder.from(new MATCH("FIELD", "VALUE"));</pre>
53   * <p>
54   * <p>Example 2 - Creating a sequence of ORs with multiple expressions (likewise for AND and XOR):
55   * <p>
56   * <pre>
57   *    FieldTextBuilder builder = FieldTexts.OR(
58   *        new MATCH("FIELD", "VALUE"),
59   *        new EQUAL("NUMERIC", 7, 8),
60   *        new EXISTS("UUID")
61   *    );</pre>
62   * <p>
63   * <p>Example 3 - Conditionally building ORs when none of the fieldtext is fixed (likewise for AND and XOR):
64   * <p>
65   * <pre>
66   *    // Create a builder suitable for appending via OR
67   *    FieldTextBuilder builder = FieldTexts.OR();
68   *
69   *    for (fieldText : fieldTextList) {
70   * builder.OR(fieldText);
71   *    }
72   *
73   *    if (builder.isMatchNothing()) {
74   *        // Nothing was added as part of the loop so the fieldtext won't match any documents
75   *    }
76   * </pre>
77   */
78  public final class FieldTextBuilder extends AbstractFieldText {

117     /**
118      * Appends the specified fieldtext onto the builder using the AND operator. Parentheses are omitted where possible
119      * and logical simplifications are made in certain cases:
120      * <p>
121      * <pre>(A AND A) {@literal =>} A</pre>
122      * <pre>(* AND A) {@literal =>} A</pre>
123      * <pre>(0 AND A) {@literal =>} 0</pre>
124      *
125      * @param fieldText A fieldtext expression or specifier.
126      * @return {@code this}
127      */
128     @Override
129     public FieldTextBuilder AND(final FieldText fieldText) {
130         Validate.notNull(fieldText, "FieldText should not be null");
131 
132         if(fieldText == this || toString().equals(fieldText.toString())) {
133             return this;
134         }
135 
136         if(isEmpty()) {
137             return setFieldText(fieldText);
138         }
139 
140         if(fieldText.isEmpty()) {
141             return this;
142         }
143 
144         if(MATCHNOTHING.equals(this) || MATCHNOTHING.equals(fieldText)) {
145             return setFieldText(MATCHNOTHING);
146         }
147 
148         return binaryOperation("AND", fieldText);
149     }

151     /**
152      * Appends the specified fieldtext onto the builder using the OR operator. Parentheses are omitted where possible
153      * and logical simplifications are made in certain cases:
154      * <p>
155      * <pre>(A OR A) {@literal =>} A</pre>
156      * <pre>(* OR A) {@literal =>} *</pre>
157      * <pre>(0 OR A) {@literal =>} A</pre>
158      *
159      * @param fieldText A fieldtext expression or specifier.
160      * @return {@code this}
161      */
162     @Override
163     public FieldTextBuilder OR(final FieldText fieldText) {
164         Validate.notNull(fieldText, "FieldText should not be null");
165 
166         if(fieldText == this || toString().equals(fieldText.toString())) {
167             return this;
168         }
169 
170         if(isEmpty() || MATCHNOTHING.equals(fieldText)) {
171             return this;
172         }
173 
174         if(fieldText.isEmpty() || MATCHNOTHING.equals(this)) {
175             return setFieldText(fieldText);
176         }
177 
178         return binaryOperation("OR", fieldText);
179     }

181     /**
182      * Appends the specified fieldtext onto the builder using the XOR operator. Parentheses are omitted where possible
183      * and logical simplifications are made in certain cases:
184      * <p>
185      * <pre>(A XOR A) {@literal =>} 0</pre>
186      * <pre>(* XOR A) {@literal =>} NOT A</pre>
187      * <pre>(0 XOR A) {@literal =>} A</pre>
188      *
189      * @param fieldText A fieldtext expression or specifier.
190      * @return {@code this}.
191      */
192     @Override
193     public FieldTextBuilder XOR(final FieldText fieldText) {
194         Validate.notNull(fieldText, "FieldText should not be null");
195 
196         if(fieldText == this || toString().equals(fieldText.toString())) {
197             return setFieldText(MATCHNOTHING);
198         }
199 
200         if(isEmpty()) {
201             return setFieldText(fieldText).NOT();
202         }
203 
204         if(fieldText.isEmpty()) {
205             return NOT();
206         }
207 
208         if(MATCHNOTHING.equals(this)) {
209             return setFieldText(fieldText);
210         }
211 
212         if(MATCHNOTHING.equals(fieldText)) {
213             return this;
214         }
215 
216         return binaryOperation("XOR", fieldText);
217     }

219     /**
220      * Appends the specified fieldtext onto the builder using the WHEN operator. A simplification is made in the case
221      * where the passed {@code fieldText} is equal to {@code this}:
222      * <p>
223      * <pre>(A WHEN A) {@literal =>} A</pre>
224      *
225      * @param fieldText A fieldtext expression or specifier.
226      * @return {@code this}
227      */
228     @Override
229     public FieldTextBuilder WHEN(final FieldText fieldText) {
230         Validate.notNull(fieldText, "FieldText should not be null");
231         Validate.isTrue(fieldText.size() >= 1, "FieldText must have a size greater or equal to 1");
232 
233         if(size() < 1) {
234             throw new IllegalStateException("Size must be greater than or equal to 1");
235         }
236 
237         // Here we assume that A WHEN A => A but is there an edge case where this doesn't work?
238         if(fieldText == this || toString().equals(fieldText.toString())) {
239             return this;
240         }
241 
242         if(MATCHNOTHING.equals(this) || MATCHNOTHING.equals(fieldText)) {
243             return setFieldText(MATCHNOTHING);
244         }
245 
246         return binaryOperation("WHEN", fieldText);
247     }

249     /**
250      * Appends the specified fieldtext onto the builder using the WHEN<i>n</i> operator.
251      *
252      * @param depth     The <i>n</i> in WHEN<i>n</i>
253      * @param fieldText A fieldtext expression or specifier.
254      * @return {@code this}.
255      */
256     @Override
257     public FieldTextBuilder WHEN(final int depth, final FieldText fieldText) {
258         Validate.isTrue(depth >= 1, "Depth must be at least 1");
259         Validate.notNull(fieldText, "FieldText should not be null");
260 
261         Validate.isTrue(fieldText.size() >= 1, "FieldText must have a size greater or equal to 1");
262 
263         if(size() < 1) {
264             throw new IllegalStateException("Size must be greater than or equal to 1");
265         }
266 
267         // We omit this 'optimization' because for large n it doesn't work, X WHENn A => 0, even if X = A
268         //if (fieldText == this || toString().equals(fieldText.toString())) {
269         //    return this;
270         //}
271         if(fieldText == this) {
272             return binaryOperation("WHEN" + depth, new FieldTextWrapper(this));
273         }
274 
275         if(MATCHNOTHING.equals(this) || MATCHNOTHING.equals(fieldText)) {
276             return setFieldText(MATCHNOTHING);
277         }
278 
279         return binaryOperation("WHEN" + depth, fieldText);
280     }

292     /**
293      * Does the work for the OR, AND, XOR and WHEN methods.
294      *
295      * @param operator
296      * @param fieldText
297      * @return
298      */
299     private FieldTextBuilder binaryOperation(final String operator, final FieldText fieldText) {
300         // This should never happen as the AND, OR, XOR and WHEN methods should already have checked this
301         Validate.isTrue(fieldText != this);
302 
303         // Optimized case when fieldText is a FieldTextBuilder
304         if(fieldText instanceof FieldTextBuilder) {
305             return binaryOp(operator, (FieldTextBuilder)fieldText);
306         }
307 
308         // Special case when we're empty
309         if(componentCount == 0) {
310             fieldTextString.append(fieldText.toString());
311 
312             if(fieldText.size() > 1) {
313                 lastOperator = "";
314             }
315         } else {
316             // Add the NOTs, parentheses and operator
317             addOperator(operator);
318 
319             // Size 1 means a single specifier, so no parentheses
320             if(fieldText.size() == 1) {
321                 fieldTextString.append(fieldText.toString());
322             } else {
323                 fieldTextString.append('(').append(fieldText.toString()).append(')');
324             }
325         }
326 
327         componentCount += fieldText.size();
328         not = false;
329 
330         return this;
331     }
332 
333     /**
334      * Does the work for the OR, AND and XOR methods in the optimized case where fieldText is a FieldTextBuilder.
335      *
336      * @param operator
337      * @param fieldText
338      * @return
339      */
340     private FieldTextBuilder binaryOp(final String operator, final FieldTextBuilder fieldText) {
341         // Special case when we're empty
342         if(componentCount == 0) {
343             // Just copy the argument
344             fieldTextString.append(fieldText.fieldTextString);
345             lastOperator = fieldText.lastOperator;
346             not = fieldText.not;
347             componentCount = fieldText.componentCount;
348         } else {
349             // Add the NOTs, parentheses and operator
350             addOperator(operator);
351 
352             if(fieldText.lastOperator == null || fieldText.lastOperator.equals(operator) || fieldText.not) {
353                 fieldTextString.append(fieldText.toString());
354             } else {
355                 fieldTextString.append('(').append(fieldText.toString()).append(')');
356             }
357 
358             componentCount += fieldText.size();
359             not = false;
360         }
361 
362         return this;
363     }

365     /**
366      * The first section of the fieldtext manipulation process, where the NOTs are handled and the operator is added,
367      * are common to both binaryOp methods so this method saves us repeating the code.
368      *
369      * @param operator
370      */
371     private void addOperator(final String operator) {
372         // This should never happen but sanity check...
373         Validate.notNull(operator);
374 
375         if(lastOperator == null) {
376             lastOperator = operator;
377         }
378 
379         /* Logic to determine whether to put parentheses around the existing fieldtext. This is tied to the use of the
380          * NOT operator as it also adds parentheses.
381          */
382         if(not) {
383             if(componentCount == 1) {
384                 fieldTextString.insert(0, "NOT+");
385             } else {
386                 fieldTextString.insert(0, "NOT(").append(')');
387             }
388         } else if(!lastOperator.equals(operator)) {
389             fieldTextString.insert(0, '(').append(')');
390         }
391 
392         lastOperator = operator;
393 
394         fieldTextString.append('+').append(operator).append('+');
395     }

397     /**
398      * Applies a NOT operator to the {@code FieldTextBuilder}. Note that two calls to {@code NOT()} will cancel out.
399      *
400      * @return {@code this}
401      */
402     @Override
403     public FieldTextBuilder NOT() {
404         // NOT * => 0
405         // NOT 0 => *
406 
407         if(isEmpty()) {
408             return setFieldText(MATCHNOTHING);
409         }
410 
411         if(MATCHNOTHING.equals(this)) {
412             return clear();
413         }
414 
415         not ^= true;
416         return this;
417     }
