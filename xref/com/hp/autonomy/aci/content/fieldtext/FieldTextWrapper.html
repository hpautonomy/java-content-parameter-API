6   package com.hp.autonomy.aci.content.fieldtext;
7   
8   import org.apache.commons.lang.Validate;
9   
10  /**
11   * The {@code FieldTextWrapper} class makes an immutable copy of another {@code FieldText} object. Though the classes
12   * representing specifiers in this package are already immutable, {@code FieldTextBuilder} is not and in general
13   * {@code FieldText} implementations are not obliged to be immutable. A {@code FieldTextWrapper} instance is not backed
14   * by the original object, a copy of its size and string representation are taken.
15   *
16   * Objects that are not {@code FieldText} instances can also be wrapped, assuming that their {@code toString()} methods
17   * return a fieldtext expression.
18   */
19  public final class FieldTextWrapper extends AbstractFieldText {
20      private final int size;
21      private final String fieldText;
22  
23      /**
24       * Creates an immutable copy of the specified {@code FieldText} object. The copy is not backed by the original
25       * object.
26       *
27       * @param fieldText The object to copy. Must be non-null.
28       */
29      public FieldTextWrapper(final FieldText fieldText) {
30          Validate.notNull(fieldText, "FieldText must not be null");
31  
32          this.fieldText = fieldText.toString();
33          this.size = fieldText.size();
34  
35          Validate.notNull(this.fieldText, "FieldText string must not be null");
36          Validate.isTrue(size >= 0, "FieldText size must be non-negative");
37      }
38  
39      /**
40       * Though primarily intended for converting a {@code String} representation of fieldtext into genuine a
41       * {@code FieldText} object, any object may be used with a suitable {@code toString()} implementation. A suitable
42       * size will be assigned though it may not accurately reflect the actual size of the fieldtext expression.
43       *
44       * @param fieldText The object to wrap. Must be non-null.
45       */
46      public FieldTextWrapper(final Object fieldText) {
47          Validate.notNull(fieldText, "FieldText must not be null");
48  
49          this.fieldText = fieldText.toString().trim();
50  
51          if (this.fieldText.isEmpty()) {
52              this.size = 0;
53          }
54          else {
55              this.size = 2;
56          }
57      }
58  
59      /**
60       * Though primarily intended for converting a {@code String} representation of fieldtext into genuine a
61       * {@code FieldText} object, any object may be used with a suitable {@code toString()} implementation.
62       *
63       * @param fieldText The object to wrap. Must be non-null.
64       * @param size The size of the {@code FieldTextWrapper}. Must be non-negative.
65       */
66      public FieldTextWrapper(final Object fieldText, final int size) {
67          Validate.notNull(fieldText, "FieldText must not be null");
68          Validate.isTrue(size >= 0, "FieldText size must not be negative");
69  
70          this.fieldText = fieldText.toString().trim();
71          this.size = size;
72  
73          Validate.notNull(this.fieldText, "FieldText must not be null");
74      }
75  
76      @Override
77      public int size() {
78          return size;
79      }
80  
81      @Override
82      public String toString() {
83          return fieldText;
84      }
85  }
