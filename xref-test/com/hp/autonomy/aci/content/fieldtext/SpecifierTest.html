137     @Test
138     public void testFinality() {
139         // Most methods of Specifier are final to ensure immutability. This test ensures that this is not accidentally
140         // changed.
141 
142         final Class<?>[] specifiers = {
143                 Specifier.class,
144                 BIAS.class,
145                 BIASVAL.class,
146                 EMPTY.class,
147                 EQUAL.class,
148                 EQUALALL.class,
149                 EXISTS.class,
150                 GREATER.class,
151                 LESS.class,
152                 MATCH.class,
153                 MATCHALL.class,
154                 MATCHRECURSE.class,
155                 NOTEQUAL.class,
156                 STRING.class,
157                 STRINGALL.class,
158                 WILD.class
159         };
160 
161         for (final Class<?> specifier : specifiers) {
162             final Method[] methods = specifier.getMethods();
163 
164             for (final Method method : methods) {
165                 final int modifiers = method.getModifiers();
166 
167                 if (Modifier.isPublic(modifiers) || Modifier.isProtected(modifiers)) {
168                     if (!Modifier.isFinal(modifiers) && !Modifier.isStatic(modifiers)) {
169                         fail("Method " + method.getName() + " of " + specifier.getSimpleName() + " is not final, static or hidden");
170                     }
171                 }
172             }
173         }
174     }

192     @SuppressWarnings("EqualsBetweenInconvertibleTypes")
193     @Test
194     public void testEqualsHashCode() {
195         final FieldText specifier1 = new Specifier("OPERATOR", "field", "val1", "val2");
196         final FieldText specifier2 = new Specifier("OPERATOR", "field", "val1", "val2");
197         final FieldText specifier3 = new Specifier("OPERATOR", "field", "val1", "val3");
198         final FieldText specifier4 = new Specifier("MATCH", "field", "val1", "val2");
199         final FieldText specifier5 = new Specifier("OPERATOR", "title", "val1", "val2");
200 
201         final FieldText builder1 = new FieldTextBuilder(new Specifier("OPERATOR", "field", "val1", "val2"));
202 
203         //noinspection EqualsWithItself
204         assertThat(specifier1.equals(specifier1), is(true));
205         assertThat(specifier1.equals(specifier2), is(true));
206         assertThat(specifier1.equals(builder1), is(true));
207 
208         assertThat(specifier1.equals(specifier3), is(false));
209         assertThat(specifier1.equals(specifier4), is(false));
210         assertThat(specifier1.equals(specifier5), is(false));
211 
212         //noinspection EqualsBetweenInconvertibleTypes
213         assertThat(specifier1.equals("OPERATOR{val1,val2}:field"), is(false));
214         //noinspection ObjectEqualsNull
215         assertThat(specifier1.equals(null), is(false));
216 
217         System.out.println("specifier1 - " + specifier1.toString() + ", hashcode - " + specifier1.hashCode());
218         System.out.println("specifier2 - " + specifier2.toString() + ", hashcode - " + specifier2.hashCode());
219         System.out.println("builder1   - " + builder1.toString() + ", hashcode - " + builder1.hashCode());
220 
221         assertThat(specifier1.hashCode(), is(equalTo(specifier1.hashCode())));
222         assertThat(specifier1.hashCode(), is(equalTo(specifier2.hashCode())));
223         assertThat(specifier1.hashCode(), is(equalTo(builder1.hashCode())));
